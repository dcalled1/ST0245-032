public int ArraySum(int[] a,int n) {
  int sum=0;
  for(int i=0;i<n && i<a.length();i++) {
    sum+=a[i];
  }
  return sum;
}

public void tablas(int n) {
  for(int i=0;i<=n;i++)
    for(int j=0;j<=n;j++)
      i*j;
}

    j=i;
    while(j>0 && a[j-1]>a[j]) {
      temp=a[j];
      a[j]=a[j-1];
      a[j-1]=temp;
      j-=1;
    }
  }
}
